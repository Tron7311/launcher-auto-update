cmake_minimum_required(VERSION 3.20)
project(LauncherApp VERSION 1.0.0 LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# Add source files - ONLY main.cpp
add_executable(LauncherApp
    src/main.cpp
)

# Include directories
target_include_directories(LauncherApp PRIVATE ${CMAKE_CURRENT_SOURCE_DIR}/src)

# Link Windows libraries
target_link_libraries(LauncherApp
    PRIVATE
    user32           # User interface for MessageBox
)

# MSVC specific settings
if(MSVC)
    target_compile_options(LauncherApp PRIVATE /W4)
    target_link_options(LauncherApp PRIVATE "/SUBSYSTEM:CONSOLE")
endif()

# Set output directory
set_target_properties(LauncherApp PROPERTIES
    RUNTIME_OUTPUT_DIRECTORY "${CMAKE_BINARY_DIR}/Release"
)
